50 ## Documented proc \c parseSystemDictionary .
51 #
52 # Parse the SALSubsystems.tcl and create a SYSDIC array object
53 # for use by salgenerator
54 #
55 proc parseSystemDictionary { } {
56 global env SYSDIC SAL_WORK_DIR OPTIONS
57  if { $OPTIONS(verbose)} {puts stdout "###TRACE>>> parseSystemDictionary"}
58  set SYSDIC(systems) ""
59  getValidGenerics
60  set fin [open $env(SAL_WORK_DIR)/SALSubsystems.xml r]
61  while { [gets $fin rec] > -1} {
62  set tag [lindex [split $rec "<>"] 1]
63  set value [lindex [split $rec "<>"] 2]
64  if { $tag == "Name"} {
65  set name $value
66  lappend SYSDIC(systems) $name
67  }
68  if { $tag == "Description"} {
69  set SYSDIC($name,Description) $value
70  }
71  if { $tag == "Generics"} {
72  if { $value == "yes"} {
73  set SYSDIC($name,hasAllGenerics) 1
74  } else {
75  if { [llength [split $value ,]] > 1} {
76  validateGenerics $name $value
77  if { $OPTIONS(verbose)} {puts stdout "TRACE------ $name has generics $value"}
78  set SYSDIC($name,hasGenerics) 1
79  set SYSDIC($name,genericsUsed) $value
80  }
81  }
82  }
83  if { $tag == "Enumeration" || $tag == "IndexEnumeration"} {
84  if { $value != "" && $value != "no"} {
85  set ids [split $value ,]
86  set SYSDIC($name,keyedID) 1
87  set SYSDIC($name,IndexEnumeration) ""
88  set idx 1
89  foreach i $ids {
90  if { [llength [split $i "="]] > 1} {
91  set idx [string trim [lindex [split $i "="] 1]]
92  set eval [string trim [lindex [split $i "="] 0]]
93  set SYSDIC($name,$idx) $eval
94  lappend SYSDIC($name,IndexEnumeration) "$idx:$eval"
95  } else {
96  set SYSDIC($name,$idx) $i
97  lappend SYSDIC($name,IndexEnumeration) "$idx:$i"
98  incr idx 1
99  }
100  }
101  }
102  }
103  if { $tag == "RuntimeLanguages"} {
104  set langs [split $value ,]
105  foreach l $langs {
106  if { $OPTIONS(verbose)} {puts stdout "TRACE------ $name needs runtime support for $l"}
107  set support [string tolower [string trim $l]]
108  set SYSDIC($name,$support) 1
109  }
110  if { [info exists SYSDIC($name,labview)]} {set SYSDIC($name,cpp) 1}
111  if { [info exists SYSDIC($name,salpy)]} {set SYSDIC($name,cpp) 1}
112  }
113  }
114  close $fin
115  set SYSDIC(systems) [lsort $SYSDIC(systems)]
116  if { $OPTIONS(verbose)} {puts stdout "###TRACE<<< parseSystemDictionary"}
117 }
