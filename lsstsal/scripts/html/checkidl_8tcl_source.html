18 ## Documented proc \c noncoding .
19 # \param[in] r Record from an IDL file
20 #
21 # Check if a record of IDL is coding a data item or not
22 # Returns 1 if not coding for a data item
23 #
24 proc noncoding { r } {
25 global NEWCONSTS KEYINDEX
26  set r [string trim $r "\{"]
27  if { [string range $r 0 5] == "#index"} {
28  set KEYINDEX [lindex $r 1]
29  return 1
30  }
31  if { [string trim $r] == "" || [string range $r 0 0] == "#" || [string range $r 0 1] == "//"} {
32  return 1
33  }
34  if { [string range $r 0 4] == "const"} {
35  set var [lindex [split [lrange $r 1 end] =] 0]
36  set val [lindex [split [lrange $r 1 end] =] 1]
37  set NEWCONSTS([string trim $var]) $val
38  return 1
39  }
40  return 0
41 }

44 ## Documented proc \c validitem .
45 # \param[in] type Data type of item
46 # \param[in] item Name of the data item
47 # \param[in] op Optional type of information to return (dim,type,id)
48 #
49 # Check the data type of an item and return the correctly
50 # formatted IDL definition, error if a reserved word is used
51 #
52 proc validitem { type item {op all} } {
53 global NEWCONSTS IDLSIZES IDLRESERVED
54  if { $type == "int"} {set type "long"}
55  if { $type == "byte"} {set type "octet"}
56  if { $type == "longlong"} {set type "long long"}
57  if { [string range $type 0 5] == "string"} {
58  set id [string trim $item "\;"]
59  set siz [lindex [split $type "<>"] 1]
60  if { [info exists NEWCONSTS($siz)]} {set siz $NEWCONSTS($siz)}
61  set nof $siz
62  set t $type
63  if { $siz != ""} {
64  set v " string<$siz> $id"
65  } else {
66  set v " string $id"
67  }
68  } else {
69  set it [split $item "\[\]"]
70  set t $type
71  if { [llength $it] == 1} {
72  set id [string trim [lindex $it 0] "\;"]
73  set siz 1
74  set v " $t $id"
75  } else {
76  set id [string trim [lindex $it 0] "\;"]
77  set siz [lindex $it 1]
78  if { [info exists NEWCONSTS($siz)]} {set siz $NEWCONSTS($siz)}
79  set v " $type $id\[$siz\]"
80  }
81  }
82  if { $type == "char"} {
83  set tid [lindex [split $id "\["] 0]
84  set v " string<$siz> $tid"
85  }
86 #puts stdout "valid is $v"
87  if { [lsearch $IDLRESERVED [string tolower $id]] > -1} {errorexit "Invalid use of IDL reserved token $id"}
88  if { $op == "dim"} {return $siz}
89  if { $op == "type"} {return $t}
90  if { $op == "id"} {return $id}
91  return $v
92 }

167 # Calculate the size of Topic data created at runtime
168 # Used to estimate EFD sizing requirements

170 proc streamsizes { } {
171 global NEWSIZES FREQUENCY PUBLISHERS
172  set stotal 0 ; set dtotal 0 ; set btotal 0
173  foreach s [lsort [array names NEWSIZES]] {
174  set payload $NEWSIZES($s)
175  set total [expr ($payload + 48) * $PUBLISHERS($s)]
176  if { [info exists FREQUENCY($s)] == 0} {set FREQUENCY($s) 0.1}
177  set persec [expr int($total * $FREQUENCY($s))]
178  set perday [expr $persec * 3600. * 24. /1024./1024.]
179  puts stdout "[format %-20s $s] size=[format %6d $total] rate=[format %6d $persec] Bytes/sec [format %5.1f $perday] Mb/day"
180  set btotal [expr $btotal + $total]
181  set stotal [expr $stotal + $persec]
182  set dtotal [expr $dtotal + $perday]
183  }
184  puts stdout "\n[format %-20s Total] size=[format %6d $btotal] rate=[format %6d [expr int($stotal)]] Bytes/sec [expr int($dtotal)] Mb/day"
