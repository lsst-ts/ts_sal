51 # \param[in] idlfile File containing IDL format data
52 # \param[in] targets List of Subsystems/CSCs to generate code for
53 #
54 # Generate SAL API code for a set of Subsystems/CSCs
55 #
56 proc genTelemetryCodes { idlfile targets } {
57 global DONE_CMDEVT OPTIONS ONEPYTHON SAL_DIR
58  if { $OPTIONS(verbose)} {stdlog "###TRACE>>> genTelemetryCodes $targets"}
59  foreach subsys $targets {
60  set spl [file rootname [split $subsys _]]
61  set base [lindex $spl 0]
62  if { [lindex $spl 1] != "command" && [lindex $spl 1] != "logevent" && [lindex $spl 1] != "ackcmd"} {
63  set name [join [lrange $spl 1 end] _]
64  if { $OPTIONS(cpp)} {
65  stdlog "Generating SAL CPP code for $subsys"
66  set result none
67  catch { set result [makesalcode $idlfile $base $name cpp] } bad
68  if { $result == "none"} {stdlog $bad}
69  if { $OPTIONS(verbose)} {stdlog $result}
70  }
71  if { $OPTIONS(java)} {
72  stdlog "Generating SAL Java code for $subsys"
73  set result none
74  catch { set result [exec make_salUtils] } bad
75  if { $result == "none"} {puts stderr $bad}
76  set result none
77  catch { set result [makesalcode $idlfile $base $name java] } bad
78  if { $result == "none"} {puts stderr $bad}
79  if { $OPTIONS(verbose)} {stdlog $result}
80  }
81  if { $OPTIONS(isocpp)} {
82  stdlog "Generating SAL ISOCPP code for $subsys"
83  set result none
84  catch { set result [makesalcode $idlfile $base $name isocpp] } bad
85  if { $result == "none"} {stdlog $bad}
86  if { $OPTIONS(verbose)} {stdlog $result}
87  }
88  if { $OPTIONS(python) && $ONEPYTHON == 0} {
89  stdlog "Generating SAL Python code for $subsys $ONEPYTHON"
90  set result none
91  catch { set result [makesalcode $idlfile $base $name python] } bad
92  if { $result == "none"} {stdlog $bad}
93  set ONEPYTHON 1
94  if { $OPTIONS(verbose)} {stdlog $result}
95  }
96  }
97  }
98  if { $OPTIONS(verbose)} {stdlog "###TRACE<<< genTelemetryCodes $targets"}
99 }
